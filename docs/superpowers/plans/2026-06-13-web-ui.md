# Cairn Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a statically-served web UI for managing Cairn PTY sessions, with a modular architecture supporting future plugin UI components.

**Architecture:** Framework-agnostic `DaemonClient` (plain TypeScript) handles WebTransport + wRPC communication. Thin Svelte 5 runes stores wrap it for reactivity. SvelteKit with `adapter-static` provides file-based routing and static output. `<cairn-terminal>` Web Component wraps wterm for terminal rendering.

**Tech Stack:** Svelte 5, SvelteKit, Deno, WebTransport, wRPC (hand-written codec), wterm + @wterm/ghostty, Vitest

**Spec:** `docs/superpowers/specs/2026-06-13-web-ui-design.md`

```typescript
// cairn-web/src/lib/client/encoding.ts

const textEncoder = new TextEncoder();
const textDecoder = new TextDecoder();

export class BinaryWriter {
  private chunks: Uint8Array[] = [];
  private buf = new Uint8Array(16);
  private pos = 0;

  private ensure(n: number): void {
    if (this.pos + n > this.buf.length) {
      this.chunks.push(this.buf.subarray(0, this.pos));
      this.buf = new Uint8Array(Math.max(n, 256));
      this.pos = 0;
    }
  }

  private writeByte(b: number): void {
    this.ensure(1);
    this.buf[this.pos++] = b;
  }

  writeU8(value: number): void {
    this.writeByte(value & 0xff);
  }

  writeU16(value: number): void {
    this.writeULEB128(value);
  }

  writeU32(value: number): void {
    this.writeULEB128(value);
  }

  writeU64(value: bigint): void {
    this.writeULEB128BigInt(value);
  }

  writeS32(value: number): void {
    this.writeSLEB128(value);
  }

  writeBool(value: boolean): void {
    this.writeByte(value ? 1 : 0);
  }

  writeString(value: string): void {
    const encoded = textEncoder.encode(value);
    this.writeULEB128(encoded.byteLength);
    this.writeBytes(encoded);
  }

  writeBytes(data: Uint8Array): void {
    this.ensure(data.length);
    if (data.length > this.buf.length - this.pos) {
      this.chunks.push(this.buf.subarray(0, this.pos));
      this.chunks.push(new Uint8Array(data));
      this.buf = new Uint8Array(256);
      this.pos = 0;
    } else {
      this.buf.set(data, this.pos);
      this.pos += data.length;
    }
  }

  writeOption<T>(value: T | null, encode: (w: BinaryWriter, v: T) => void): void {
    if (value === null) {
      this.writeByte(0);
    } else {
      this.writeByte(1);
      encode(this, value);
    }
  }

  writeList<T>(items: T[], encode: (w: BinaryWriter, v: T) => void): void {
    this.writeULEB128(items.length);
    for (const item of items) {
      encode(this, item);
    }
  }

  writeResult<T, E>(
    value: { ok: T } | { err: E },
    encodeOk: (w: BinaryWriter, v: T) => void,
    encodeErr: (w: BinaryWriter, v: E) => void,
  ): void {
    if ('ok' in value) {
      this.writeByte(0);
      encodeOk(this, value.ok);
    } else {
      this.writeByte(1);
      encodeErr(this, value.err);
    }
  }

  private writeULEB128(value: number): void {
    do {
      let byte = value & 0x7f;
      value >>>= 7;
      if (value !== 0) byte |= 0x80;
      this.writeByte(byte);
    } while (value !== 0);
  }

  private writeULEB128BigInt(value: bigint): void {
    do {
      let byte = Number(value & 0x7fn);
      value >>= 7n;
      if (value !== 0n) byte |= 0x80;
      this.writeByte(byte);
    } while (value !== 0n);
  }

  private writeSLEB128(value: number): void {
    let more = true;
    while (more) {
      let byte = value & 0x7f;
      value >>= 7;
      if ((value === 0 && (byte & 0x40) === 0) || (value === -1 && (byte & 0x40) !== 0)) {
        more = false;
      } else {
        byte |= 0x80;
      }
      this.writeByte(byte);
    }
  }

  finish(): Uint8Array {
    this.chunks.push(this.buf.subarray(0, this.pos));
    const total = this.chunks.reduce((sum, c) => sum + c.length, 0);
    const result = new Uint8Array(total);
    let offset = 0;
    for (const chunk of this.chunks) {
      result.set(chunk, offset);
      offset += chunk.length;
    }
    return result;
  }
}

export class BinaryReader {
  private data: Uint8Array;
  private pos: number;

  constructor(data: Uint8Array, offset = 0) {
    this.data = data;
    this.pos = offset;
  }

  get offset(): number {
    return this.pos;
  }

  get remaining(): number {
    return this.data.length - this.pos;
  }

  readU8(): number {
    return this.data[this.pos++];
  }

  readU16(): number {
    return this.readULEB128();
  }

  readU32(): number {
    return this.readULEB128();
  }

  readU64(): bigint {
    return this.readULEB128BigInt();
  }

  readS32(): number {
    return this.readSLEB128();
  }

  readBool(): boolean {
    return this.readU8() !== 0;
  }

  readString(): string {
    const len = this.readULEB128();
    const bytes = this.data.subarray(this.pos, this.pos + len);
    this.pos += len;
    return textDecoder.decode(bytes);
  }

  readBytes(len: number): Uint8Array {
    const bytes = this.data.slice(this.pos, this.pos + len);
    this.pos += len;
    return bytes;
  }

  readOption<T>(decode: (r: BinaryReader) => T): T | null {
    const tag = this.readU8();
    if (tag === 0) return null;
    return decode(this);
  }

  readList<T>(decode: (r: BinaryReader) => T): T[] {
    const len = this.readULEB128();
    const items: T[] = [];
    for (let i = 0; i < len; i++) {
      items.push(decode(this));
    }
    return items;
  }

  readResult<T, E>(
    decodeOk: (r: BinaryReader) => T,
    decodeErr: (r: BinaryReader) => E,
  ): T {
    const tag = this.readU8();
    if (tag === 0) return decodeOk(this);
    throw decodeErr(this);
  }

  private readULEB128(): number {
    let result = 0;
    let shift = 0;
    let byte: number;
    do {
      byte = this.data[this.pos++];
      result |= (byte & 0x7f) << shift;
      shift += 7;
    } while (byte & 0x80);
    return result >>> 0;
  }

  private readULEB128BigInt(): bigint {
    let result = 0n;
    let shift = 0n;
    let byte: number;
    do {
      byte = this.data[this.pos++];
      result |= BigInt(byte & 0x7f) << shift;
      shift += 7n;
    } while (byte & 0x80);
    return result;
  }

  private readSLEB128(): number {
    let result = 0;
    let shift = 0;
    let byte: number;
    do {
      byte = this.data[this.pos++];
      result |= (byte & 0x7f) << shift;
      shift += 7;
    } while (byte & 0x80);
    if (shift < 32 && (byte & 0x40) !== 0) {
      result |= -(1 << shift);
    }
    return result;
  }
}
```

Implements the wRPC invocation frame format and path-indexed response frames. This is the byte-level protocol the browser uses to call daemon functions over WebTransport.

**Wire format reference** (from `wrpc-transport` Rust crate):

Invocation frame (client → server):
```
[0x00]                           protocol version
[LEB128 len][UTF-8 bytes]       instance name (e.g., "cairn:daemon/sessions@0.1.0")
[LEB128 len][UTF-8 bytes]       function name (e.g., "list_all")
[LEB128 len][encoded params]    parameter bytes
```

Response/stream frame (bidirectional, after invocation):
```
[LEB128 path_len][LEB128 path_elem]*   path index ([] for sync, [0] for first async)
[LEB128 data_len][data bytes]           payload
```

```typescript
// cairn-web/src/lib/client/wrpc.ts

import { BinaryWriter, BinaryReader } from './encoding.js';

const PROTOCOL_VERSION = 0x00;

export interface WrpcFrame {
  path: number[];
  data: Uint8Array;
}

export function encodeInvocation(
  instance: string,
  func: string,
  params: Uint8Array,
): Uint8Array {
  const w = new BinaryWriter();
  w.writeU8(PROTOCOL_VERSION);
  w.writeString(instance);
  w.writeString(func);
  w.writeU64(BigInt(params.byteLength));
  w.writeBytes(params);
  return w.finish();
}

export function encodeFrame(path: number[], data: Uint8Array): Uint8Array {
  const w = new BinaryWriter();
  w.writeU32(path.length);
  for (const idx of path) {
    w.writeU32(idx);
  }
  w.writeU64(BigInt(data.byteLength));
  w.writeBytes(data);
  return w.finish();
}

export function decodeFrame(r: BinaryReader): WrpcFrame {
  const pathLen = r.readU32();
  const path: number[] = [];
  for (let i = 0; i < pathLen; i++) {
    path.push(r.readU32());
  }
  const dataLen = Number(r.readU64());
  const data = r.readBytes(dataLen);
  return { path, data };
}

export async function writeToStream(
  writer: WritableStreamDefaultWriter<Uint8Array>,
  data: Uint8Array,
): Promise<void> {
  await writer.write(data);
}

export async function readAllFromStream(
  reader: ReadableStreamDefaultReader<Uint8Array>,
): Promise<Uint8Array> {
  const chunks: Uint8Array[] = [];
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    chunks.push(value);
  }
  const total = chunks.reduce((sum, c) => sum + c.length, 0);
  const result = new Uint8Array(total);
  let offset = 0;
  for (const chunk of chunks) {
    result.set(chunk, offset);
    offset += chunk.length;
  }
  return result;
}

export class StreamFrameReader {
  private buffer = new Uint8Array(0);
  private reader: ReadableStreamDefaultReader<Uint8Array>;

  constructor(readable: ReadableStream<Uint8Array>) {
    this.reader = readable.getReader();
  }

  async nextFrame(): Promise<WrpcFrame | null> {
    while (true) {
      if (this.buffer.length > 0) {
        try {
          const r = new BinaryReader(this.buffer);
          const frame = decodeFrame(r);
          this.buffer = this.buffer.subarray(r.offset);
          return frame;
        } catch {
          // not enough data yet, read more
        }
      }

      const { done, value } = await this.reader.read();
      if (done) return null;

      const merged = new Uint8Array(this.buffer.length + value.length);
      merged.set(this.buffer);
      merged.set(value, this.buffer.length);
      this.buffer = merged;
    }
  }

  release(): void {
    this.reader.releaseLock();
  }
}
```

```typescript
// cairn-web/src/lib/client/connection.ts

import type { ConnectionState } from './types.js';
import type { AuthStrategy } from './auth.js';
import { TailscaleAuth } from './auth.js';

export interface ConnectionOptions {
  endpoint: string;
  certHash?: string;
  auth?: AuthStrategy;
  onStateChange?: (state: ConnectionState) => void;
}

export class CairnConnection {
  private transport: WebTransport | null = null;
  private options: ConnectionOptions;
  private state: ConnectionState = 'disconnected';
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private reconnectDelay = 100;
  private shouldReconnect = true;

  constructor(options: ConnectionOptions) {
    this.options = {
      auth: new TailscaleAuth(),
      ...options,
    };
  }

  get connectionState(): ConnectionState {
    return this.state;
  }

  async connect(): Promise<void> {
    this.shouldReconnect = true;
    await this.doConnect();
  }

  async disconnect(): Promise<void> {
    this.shouldReconnect = false;
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
      this.reconnectTimer = null;
    }
    if (this.transport) {
      this.transport.close();
      this.transport = null;
    }
    this.setState('disconnected');
  }

  async openBidiStream(): Promise<WebTransportBidirectionalStream> {
    if (!this.transport) {
      throw new Error('Not connected');
    }
    return await this.transport.createBidirectionalStream();
  }

  private async doConnect(): Promise<void> {
    this.setState('connecting');

    try {
      const options: WebTransportOptions = {};
      if (this.options.certHash) {
        options.serverCertificateHashes = [{
          algorithm: 'sha-256',
          value: hexToBuffer(this.options.certHash),
        }];
      }

      this.transport = new WebTransport(this.options.endpoint, options);
      await this.transport.ready;
      await this.options.auth!.authenticate(this.transport);

      this.setState('connected');
      this.reconnectDelay = 100;

      this.transport.closed.then(() => {
        this.transport = null;
        if (this.shouldReconnect) {
          this.scheduleReconnect();
        } else {
          this.setState('disconnected');
        }
      });
    } catch (err) {
      this.transport = null;
      this.setState('error');
      if (this.shouldReconnect) {
        this.scheduleReconnect();
      }
    }
  }

  private scheduleReconnect(): void {
    this.setState('connecting');
    this.reconnectTimer = setTimeout(() => {
      this.reconnectTimer = null;
      this.doConnect();
    }, this.reconnectDelay);
    this.reconnectDelay = Math.min(this.reconnectDelay * 2, 10_000);
  }

  private setState(state: ConnectionState): void {
    if (this.state !== state) {
      this.state = state;
      this.options.onStateChange?.(state);
    }
  }
}

function hexToBuffer(hex: string): ArrayBuffer {
  const clean = hex.replace(/[^0-9a-fA-F]/g, '');
  const bytes = new Uint8Array(clean.length / 2);
  for (let i = 0; i < bytes.length; i++) {
    bytes[i] = parseInt(clean.slice(i * 2, i * 2 + 2), 16);
  }
  return bytes.buffer;
}
```

```typescript
// cairn-web/src/lib/client/sessions.ts

import { BinaryWriter, BinaryReader } from './encoding.js';
import * as codec from './codec.js';
import { encodeInvocation, StreamFrameReader, encodeFrame } from './wrpc.js';
import type { CairnConnection } from './connection.js';
import type {
  SessionId, SessionInfo, SessionSpec, Signal,
  LogWindow, AttachInit, ClientEvent, ServerEvent,
  ExitStatus, CallContext,
} from './types.js';

const INSTANCE = 'cairn:daemon/sessions@0.1.0';

function encodeCtx(ctx: CallContext | null): BinaryWriter {
  const w = new BinaryWriter();
  w.writeOption(ctx, (w2, v) => codec.writeCallContext(w2, v));
  return w;
}

export class SessionsApi {
  constructor(private conn: CairnConnection) {}

  async listAll(ctx: CallContext | null = null): Promise<SessionInfo[]> {
    const params = encodeCtx(ctx).finish();
    return await this.invokeUnary('list-all', params, (r) =>
      r.readList((r2) => codec.readSessionInfo(r2))
    );
  }

  async inspect(id: SessionId, ctx: CallContext | null = null): Promise<SessionInfo> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    return await this.invokeUnaryResult('inspect', w.finish(), (r) => codec.readSessionInfo(r));
  }

  async create(spec: SessionSpec, ctx: CallContext | null = null): Promise<SessionInfo> {
    const w = encodeCtx(ctx);
    codec.writeSessionSpec(w, spec);
    return await this.invokeUnaryResult('create', w.finish(), (r) => codec.readSessionInfo(r));
  }

  async rename(id: SessionId, newName: string, ctx: CallContext | null = null): Promise<void> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    w.writeString(newName);
    await this.invokeVoidResult('rename', w.finish());
  }

  async restart(id: SessionId, force: boolean, ctx: CallContext | null = null): Promise<void> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    w.writeBool(force);
    await this.invokeVoidResult('restart', w.finish());
  }

  async kill(id: SessionId, signal: Signal, graceMs: number | null = null, ctx: CallContext | null = null): Promise<void> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    codec.writeSignal(w, signal);
    w.writeOption(graceMs, (w2, v) => w2.writeU32(v));
    await this.invokeVoidResult('kill', w.finish());
  }

  async kick(id: SessionId, clientId: string | null = null, ctx: CallContext | null = null): Promise<void> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    w.writeOption(clientId, (w2, v) => w2.writeString(v));
    await this.invokeVoidResult('kick', w.finish());
  }

  async *logs(
    id: SessionId,
    window: LogWindow,
    follow: boolean,
    ctx: CallContext | null = null,
  ): AsyncGenerator<Uint8Array> {
    const w = encodeCtx(ctx);
    w.writeString(id);
    codec.writeLogWindow(w, window);
    w.writeBool(follow);

    const bidi = await this.conn.openBidiStream();
    const writer = bidi.writable.getWriter();
    await writer.write(encodeInvocation(INSTANCE, 'logs', w.finish()));
    await writer.close();

    const frameReader = new StreamFrameReader(bidi.readable);
    try {
      while (true) {
        const frame = await frameReader.nextFrame();
        if (!frame) break;
        const r = new BinaryReader(frame.data);
        yield new Uint8Array(r.readList((r2) => r2.readU8()));
      }
    } finally {
      frameReader.release();
    }
  }

  async wait(id: SessionId, ctx: CallContext | null = null): Promise<ExitStatus> {
    const w = encodeCtx(ctx);
    w.writeString(id);

    const bidi = await this.conn.openBidiStream();
    const writer = bidi.writable.getWriter();
    await writer.write(encodeInvocation(INSTANCE, 'wait', w.finish()));
    await writer.close();

    const frameReader = new StreamFrameReader(bidi.readable);
    try {
      const frame = await frameReader.nextFrame();
      if (!frame) throw new Error('No response from wait');
      const r = new BinaryReader(frame.data);
      return codec.readExitStatus(r);
    } finally {
      frameReader.release();
    }
  }

  attach(
    id: SessionId,
    init: AttachInit,
    ctx: CallContext | null = null,
  ): AttachHandle {
    return new AttachHandle(this.conn, id, init, ctx);
  }

  private async invokeUnary<T>(
    func: string,
    params: Uint8Array,
    decode: (r: BinaryReader) => T,
  ): Promise<T> {
    const bidi = await this.conn.openBidiStream();
    const writer = bidi.writable.getWriter();
    await writer.write(encodeInvocation(INSTANCE, func, params));
    await writer.close();

    const frameReader = new StreamFrameReader(bidi.readable);
    try {
      const frame = await frameReader.nextFrame();
      if (!frame) throw new Error(`No response from ${func}`);
      return decode(new BinaryReader(frame.data));
    } finally {
      frameReader.release();
    }
  }

  private async invokeUnaryResult<T>(
    func: string,
    params: Uint8Array,
    decodeOk: (r: BinaryReader) => T,
  ): Promise<T> {
    return await this.invokeUnary(func, params, (r) => {
      return r.readResult(
        decodeOk,
        (r2) => {
          const err = codec.readCairnError(r2);
          return new Error(`${err.code}: ${err.message}`);
        },
      );
    });
  }

  private async invokeVoidResult(
    func: string,
    params: Uint8Array,
  ): Promise<void> {
    await this.invokeUnary(func, params, (r) => {
      r.readResult(
        () => {},
        (r2) => {
          const err = codec.readCairnError(r2);
          return new Error(`${err.code}: ${err.message}`);
        },
      );
    });
  }
}

export class AttachHandle {
  private bidi: WebTransportBidirectionalStream | null = null;
  private writer: WritableStreamDefaultWriter<Uint8Array> | null = null;
  private frameReader: StreamFrameReader | null = null;

  constructor(
    private conn: CairnConnection,
    private id: SessionId,
    private init: AttachInit,
    private ctx: CallContext | null,
  ) {}

  async start(): Promise<AsyncGenerator<ServerEvent>> {
    const w = new BinaryWriter();
    w.writeOption(this.ctx, (w2, v) => codec.writeCallContext(w2, v));
    w.writeString(this.id);
    codec.writeAttachInit(w, this.init);

    this.bidi = await this.conn.openBidiStream();
    this.writer = this.bidi.writable.getWriter();
    await this.writer.write(encodeInvocation(INSTANCE, 'attach', w.finish()));

    this.frameReader = new StreamFrameReader(this.bidi.readable);
    return this.readEvents();
  }

  async sendEvent(event: ClientEvent): Promise<void> {
    if (!this.writer) throw new Error('Not attached');
    const ew = new BinaryWriter();
    codec.writeClientEvent(ew, event);
    await this.writer.write(encodeFrame([0], ew.finish()));
  }

  async detach(): Promise<void> {
    await this.sendEvent({ tag: 'detach' });
    this.cleanup();
  }

  private async *readEvents(): AsyncGenerator<ServerEvent> {
    if (!this.frameReader) return;
    try {
      while (true) {
        const frame = await this.frameReader.nextFrame();
        if (!frame) break;
        const r = new BinaryReader(frame.data);
        yield codec.readServerEvent(r);
      }
    } finally {
      this.cleanup();
    }
  }

  private cleanup(): void {
    this.writer?.releaseLock();
    this.frameReader?.release();
    this.writer = null;
    this.frameReader = null;
    this.bidi = null;
  }
}
```

Note: the `invokeUnaryResult` method for `result<T, error>` return types is a placeholder — the exact decoding depends on whether the server encodes the sync result as a path-indexed frame or directly. This will need to be validated against the real daemon and adjusted if the frame format differs. The key invariant is that CairnError is thrown as an Error, not returned.

This is the core interactive feature. `Terminal.svelte` wraps wterm with the `@wterm/ghostty` backend and manages the attach lifecycle. It's designed to be compiled to a `<cairn-terminal>` Web Component in Task 13.

**wterm API reference:**
- `new WTerm(element, { cols, rows, core })` — creates terminal in a container element
- `await term.init()` — loads WASM, initializes rendering
- `term.write(data)` — feeds bytes to the terminal
- `term.onData = (data) => {}` — callback for user input
- `term.onResize = (cols, rows) => {}` — callback for resize events
- `term.resize(cols, rows)` — manual resize
- `term.destroy()` — cleanup
- `GhosttyCore.load()` — loads the ghostty WASM backend

```svelte
<script lang="ts">
  import { onMount, onDestroy } from 'svelte';
  import type { DaemonClient } from '$lib/client/client.js';
  import type { AttachHandle } from '$lib/client/sessions.js';
  import type { ServerEvent, ExitStatus } from '$lib/client/types.js';

  let {
    sessionId,
    client,
    fontSize = 14,
    fontFamily = "'Berkeley Mono', 'JetBrains Mono', 'Fira Code', monospace",
    onattached = () => {},
    ondetached = (_reason: string) => {},
    onexited = (_exit: ExitStatus) => {},
  } = $props();

  let container: HTMLDivElement;
  let term: any = null;
  let attachHandle: AttachHandle | null = null;
  let exitStatus = $state<ExitStatus | null>(null);
  let resizeObserver: ResizeObserver | null = null;

  onMount(async () => {
    const [{ WTerm }, { GhosttyCore }] = await Promise.all([
      import('@wterm/dom'),
      import('@wterm/ghostty'),
    ]);

    const core = await GhosttyCore.load();

    term = new WTerm(container, {
      cols: 80,
      rows: 24,
      core,
    });
    await term.init();

    term.onData = (data: string) => {
      const bytes = new TextEncoder().encode(data);
      attachHandle?.sendEvent({ tag: 'input', value: bytes });
    };

    resizeObserver = new ResizeObserver(() => {
      if (term && container) {
        const cols = term.cols;
        const rows = term.rows;
        attachHandle?.sendEvent({ tag: 'resize', value: [cols, rows] });
      }
    });
    resizeObserver.observe(container);

    await startAttach();
  });

  onDestroy(() => {
    resizeObserver?.disconnect();
    attachHandle?.detach().catch(() => {});
    term?.destroy();
  });

  async function startAttach() {
    if (!client || !sessionId) return;

    const cols = term?.cols ?? 80;
    const rows = term?.rows ?? 24;

    attachHandle = client.sessions.attach(sessionId, {
      cols,
      rows,
      noStdin: false,
    });

    try {
      const events = await attachHandle.start();
      onattached();

      for await (const event of events) {
        handleServerEvent(event);
      }
    } catch (err) {
      ondetached(err instanceof Error ? err.message : 'connection lost');
    }
  }

  function handleServerEvent(event: ServerEvent) {
    switch (event.tag) {
      case 'snapshot':
        term?.write(event.value);
        break;
      case 'output':
        term?.write(event.value);
        break;
      case 'exited':
        exitStatus = event.value;
        onexited(event.value);
        break;
      case 'error':
        ondetached(`${event.value.code}: ${event.value.message}`);
        break;
    }
  }
</script>
```

Extract a `Transport` interface so `CairnConnection` is not coupled to the `WebTransport` browser API. This enables integration testing over Unix sockets and opens the door for future transport backends (WebSocket, etc.).

The abstraction covers the "open a bidirectional byte stream" operation — everything above it (wRPC framing, WIT codecs, DaemonClient API) is transport-agnostic. Everything below it (WebTransport connection lifecycle, cert hashing, QUIC reconnection) is transport-specific and only testable in its native environment.

- [ ] **Step 1: Define the Transport interface**

- [ ] **Step 2: Implement WebTransportTransport**

```typescript
export class WebTransportTransport implements Transport {
  private wt: WebTransport;

  constructor(endpoint: string, certHash?: string) {
    const options: WebTransportOptions = {};
    if (certHash) {
      options.serverCertificateHashes = [{
        algorithm: 'sha-256',
        value: hexToBuffer(certHash),
      }];
    }
    this.wt = new WebTransport(endpoint, options);
  }

  get ready(): Promise<void> {
    return this.wt.ready.then(() => {});
  }

  get closed(): Promise<void> {
    return this.wt.closed.then(() => {});
  }

  async openBidiStream(): Promise<BidiStream> {
    return await this.wt.createBidirectionalStream();
  }

  close(): void {
    this.wt.close();
  }
}

function hexToBuffer(hex: string): ArrayBuffer {
  const clean = hex.replace(/[^0-9a-fA-F]/g, '');
  const bytes = new Uint8Array(clean.length / 2);
  for (let i = 0; i < bytes.length; i++) {
    bytes[i] = parseInt(clean.slice(i * 2, i * 2 + 2), 16);
  }
  return bytes.buffer;
}
```

```typescript
// Not used in the browser build.

import type { Transport, BidiStream } from './transport.js';

export class UnixSocketTransport implements Transport {
  constructor(private socketPath: string) {}

  async openBidiStream(): Promise<BidiStream> {
    const conn = await Deno.connect({
      transport: 'unix',
      path: this.socketPath,
    });

    return {
      readable: conn.readable,
      writable: conn.writable,
    };
  }

  close(): void {
    // Unix socket transport is connectionless (new conn per invocation),

  }
}
```

```typescript
// cairn-web/tests/integration/harness.ts
//
// Starts a real cairn-daemon on a temp directory Unix socket.
// Mirrors the Rust DaemonHarness in cairn-daemon/tests/common/mod.rs.

import { join } from 'node:path';
import { mkdtempSync } from 'node:fs';
import { tmpdir } from 'node:os';

export interface DaemonHandle {
  socketPath: string;
  process: Deno.ChildProcess;
  cleanup: () => Promise<void>;
}

export async function startDaemon(): Promise<DaemonHandle> {
  const dir = mkdtempSync(join(tmpdir(), 'cairn-test-'));
  const socketPath = join(dir, 'cairn.sock');

  const command = new Deno.Command('cargo', {
    args: [
      'run', '-p', 'cairn-daemon', '--',
      '--socket', socketPath,
    ],
    stdout: 'piped',
    stderr: 'piped',
  });

  const process = command.spawn();

  // Wait for the socket to appear (daemon is ready)
  const deadline = Date.now() + 10_000;
  while (Date.now() < deadline) {
    try {
      await Deno.stat(socketPath);
      break;
    } catch {
      await new Promise((r) => setTimeout(r, 100));
    }
  }

  // Verify socket exists
  await Deno.stat(socketPath);

  return {
    socketPath,
    process,
    cleanup: async () => {
      process.kill('SIGTERM');
      await process.status;
      await Deno.remove(dir, { recursive: true });
    },
  };
}
```

Note: the exact CLI flags for specifying the socket path depend on the daemon's argument parsing. Check `cairn-daemon/src/main.rs` for the actual flag names and adjust. The harness pattern mirrors `cairn-daemon/tests/common/mod.rs`.

```typescript
    expect(inspected.name).toBe('new-name');
  });

  it('streams logs from a session', async () => {
    const spec: SessionSpec = {
      name: 'logs-test',
      command: ['echo', 'hello world'],
      env: [],
      envInherit: true,
      workdir: null,
      tty: false,
      stdin: false,
      idleTimeoutSecs: null,
      scrollbackLines: 1000,
    };

    const created = await client.sessions.create(spec);

    // Wait for echo to finish
    await client.sessions.wait(created.id);

    const chunks: Uint8Array[] = [];
    for await (const chunk of client.sessions.logs(created.id, { tag: 'all' }, false)) {
      chunks.push(chunk);
    }

    const output = new TextDecoder().decode(
      new Uint8Array(chunks.flatMap((c) => Array.from(c)))
    );
    expect(output).toContain('hello world');
  });
});
```
